\newcount\NCKUDeprecatedErrorCount
\newcount\NCKUDeprecatedStopCount
\NCKUDeprecatedErrorCount=0
\NCKUDeprecatedStopCount=0

\let\NCKUOriginalErrmessage\errmessage
\let\NCKUOriginalStop\stop
\def\NCKUCurrentCommand{unset}
\def\NCKUExpectedMessage{}

\long\def\errmessage#1{%
  \advance\NCKUDeprecatedErrorCount by 1%
  \def\NCKUActualMessage{#1}%
  \ifx\NCKUActualMessage\NCKUExpectedMessage
    \typeout{NCKU-DEPRECATED-ERROR-PASS: \NCKUCurrentCommand}%
  \else
    \NCKUOriginalErrmessage{Deprecated diagnostic changed for \NCKUCurrentCommand}%
  \fi
}
\def\stop{%
  \advance\NCKUDeprecatedStopCount by 1%
  \typeout{NCKU-DEPRECATED-STOP-PASS: \NCKUCurrentCommand}%
}

\long\def\NCKUCheckDeprecated#1#2#3{%
  \def\NCKUCurrentCommand{#2}%
  \def\NCKUExpectedMessage{#3}%
  #1%
}
