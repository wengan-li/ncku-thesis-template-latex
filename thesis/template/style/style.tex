% Institution style profile selection.
%
% The repository default is NCKU. A non-NCKU template maintainer changes this
% value (or defines it before template/configure.tex) and maintains the profile
% under template/style/<name>/<name>.tex. Exactly one profile is loaded.
\providecommand{\TemplateStyleName}{ncku}
\input{./template/style/base/base}
\edef\TemplateStyleFile{./template/style/\TemplateStyleName/\TemplateStyleName.tex}
\IfFileExists{\TemplateStyleFile}
  {\input{\TemplateStyleFile}}
  {\errmessage{Template style file '\TemplateStyleFile' was not found}}

\ifdefstring{\VarActiveTemplateStyle}{unregistered}
  {\errmessage{Template style '\TemplateStyleName' did not register itself}}
  {}
