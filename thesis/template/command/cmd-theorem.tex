\newcommand{\SetTheoremContentLabel}[1][\empty]
{%
  \ifthenelse{\equal{#1}{\empty}}{}{\IfNoValueF{#1}{\label{#1}}}%
} % End of \DeclareDocumentCommand{}

\newcommand{\InsertTheoremContent}[3][\empty]
{%
  % Parse the input
  \NCKUPrivateSetInsertTheoremKeys{#1}%
  %
  \ifthenelse{\equal{\TmpValueTitle}{\empty}}%
  {%
    \begin{#2}%
  }%
  {%
    \begin{#2}[\TmpValueTitle]%
  }%
  % Freeze the optional title before \label writes nameref metadata. Without this
  % copy the auxiliary file stores the mutable \TmpValueTitle control sequence,
  % which is reset by the next insertion.
  \ifthenelse{\equal{\TmpValueTitle}{\empty}}{}%
    {\expandafter\let\csname @currentlabelname\endcsname\TmpValueTitle}%
  %
  \SetTheoremContentLabel[\TmpValueLabel]%
  %
  #3%
  %
  \ifthenelse{\equal{#2}{\GetTheoremProofFormatEnvironmentName}}%
    {$\quad\blacksquare$}{}%
  %
  \end{#2}%
} % End of \newcommand{}

      { \keys_set:nn { ncku / theorem-format } {#2} }

% Counter values can form chains (for example Problem -> Theorem -> Definition
% -> empty). Resolve the complete chain before redefining the source getter so
% forward references and self-lookups never leave mutable aliases behind.
\newcount\NCKUPrivateTheoremCounterDepth

\newcommand{\NCKUPrivateResolveTheoremCounter}[1]
{%
  \def\NCKUPrivateTheoremCounterOrigin{#1}%
  \def\NCKUPrivateResolvedTheoremCounter{#1}%
  \NCKUPrivateTheoremCounterDepth=0\relax
  \NCKUPrivateResolveTheoremCounterStep
} % End of \newcommand{}

\newcommand{\NCKUPrivateResolveTheoremCounterStep}
{%
  \ifthenelse{\equal{\NCKUPrivateResolvedTheoremCounter}{Section}}%
  {%
    \def\NCKUPrivateResolvedTheoremCounter{section}%
  }%
  {%
    \ifthenelse{\equal{\NCKUPrivateResolvedTheoremCounter}{section}}%
    {}%
    {%
      \ifthenelse{\equal{\NCKUPrivateResolvedTheoremCounter}{\empty}}%
      {}%
      {%
        \ifcsname NCKUPrivateTheoremFormatRegistered@\NCKUPrivateResolvedTheoremCounter\endcsname
          \advance\NCKUPrivateTheoremCounterDepth by 1\relax
          \ifnum\NCKUPrivateTheoremCounterDepth>21\relax
            \PackageError{ncku-thesis-template}{Cyclic theorem counter configuration starting at `\NCKUPrivateTheoremCounterOrigin'}{Use Section, empty, or an acyclic registered theorem type.}%
            \def\NCKUPrivateResolvedTheoremCounter{\empty}%
          \else
            \expandafter\let\expandafter\NCKUPrivateResolvedTheoremCounter
              \csname GetTheorem\NCKUPrivateResolvedTheoremCounter FormatFollowCounter\endcsname
            \NCKUPrivateResolveTheoremCounterStep
          \fi
        \else
          % Preserve arbitrary existing LaTeX counters as terminal values. Only
          % registered theorem names are traversed recursively.
        \fi
      }%
    }%
  }%
} % End of \newcommand{}

\newcommand{\MappingTheoremCounterResetCounter}[2][\empty]
{%
  \ifcsname NCKUPrivateTheoremFormatRegistered@#1\endcsname
    \expandafter\edef\csname GetTheorem#1FormatFollowCounter\endcsname{#2}%
  \fi
} % End of \newcommand{}

\newcommand{\MappingTheoremCounter}[2][\empty]
{%
  \ifthenelse{\equal{#2}{Section}}%
  {%
    \NCKUPrivateResolveTheoremCounter{#2}%
    \MappingTheoremCounterResetCounter[#1]{\NCKUPrivateResolvedTheoremCounter}%
  }%
  {%
    \ifcsname NCKUPrivateTheoremFormatRegistered@#2\endcsname
      \NCKUPrivateResolveTheoremCounter{#2}%
      \MappingTheoremCounterResetCounter[#1]{\NCKUPrivateResolvedTheoremCounter}%
    \fi
  }%
} % End of \newcommand{}

\newcommand{\NCKUPrivateLoadTheoremFormatState}[1]
{%
  \expandafter\let\expandafter\NCKUPrivateTheoremStyle
    \csname NCKUPrivateTheoremStyle@#1\endcsname
  \expandafter\let\expandafter\NCKUPrivateTheoremPolicy
    \csname NCKUPrivateTheoremPolicy@#1\endcsname
  \expandafter\let\expandafter\NCKUPrivateTheoremEnvironmentName
    \csname GetTheorem#1FormatEnvironmentName\endcsname
  \expandafter\let\expandafter\NCKUPrivateTheoremShowText
    \csname GetTheorem#1FormatShowText\endcsname
  \expandafter\let\expandafter\NCKUPrivateTheoremFollowCounter
    \csname GetTheorem#1FormatFollowCounter\endcsname
} % End of \newcommand{}

\newcommand{\NCKUPrivateCreateNumberedTheorem}
{%
  % Freeze registry state into the environment definition. Reusing the private
  % scratch getters directly would make every heading follow the final row.
  \ifthenelse{\equal{\NCKUPrivateTheoremFollowCounter}{\empty}}%
  {%
    \edef\NCKUPrivateCreateTheoremCommand{%
      \noexpand\newtheorem{\NCKUPrivateTheoremEnvironmentName}{\NCKUPrivateTheoremShowText}}%
  }%
  {%
    \edef\NCKUPrivateCreateTheoremCommand{%
      \noexpand\newtheorem{\NCKUPrivateTheoremEnvironmentName}{\NCKUPrivateTheoremShowText}[\NCKUPrivateTheoremFollowCounter]}%
  }%
  \NCKUPrivateCreateTheoremCommand
} % End of \newcommand{}

\newcommand{\NCKUPrivateCreateUnnumberedTheorem}
{%
  \edef\NCKUPrivateCreateTheoremCommand{%
    \noexpand\newtheorem*{\NCKUPrivateTheoremEnvironmentName}{\NCKUPrivateTheoremShowText}}%
  \NCKUPrivateCreateTheoremCommand
} % End of \newcommand{}

\newcommand{\NCKUPrivateInitRegisteredTheoremFormat}[1]
{%
  \NCKUPrivateLoadTheoremFormatState{#1}%
  \expandafter\theoremstyle\expandafter{\NCKUPrivateTheoremStyle}%
  \ifthenelse{\equal{\NCKUPrivateTheoremFollowCounter}{\empty}}%
  {%
    \ifthenelse{\equal{\NCKUPrivateTheoremPolicy}{optional}}%
    {%
      \NCKUPrivateCreateUnnumberedTheorem
    }%
    {%
      \NCKUPrivateCreateNumberedTheorem
    }%
  }%
  {%
    \MappingTheoremCounter[#1]{\NCKUPrivateTheoremFollowCounter}%
    \NCKUPrivateLoadTheoremFormatState{#1}%
    \NCKUPrivateCreateNumberedTheorem
  }%
} % End of \newcommand{}
