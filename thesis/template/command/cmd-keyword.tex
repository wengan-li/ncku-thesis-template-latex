% Common function
\def\AppendKeywordString#1#2{\edef#1{#1#2}}
\newcommand{\AppendOptionalKeywords}[9]
{%
  \ifthenelse{\equal{#2}{\empty}}{}{\AppendKeywordString{#1}{, #2}}%
  \ifthenelse{\equal{#3}{\empty}}{}{\AppendKeywordString{#1}{, #3}}%
  \ifthenelse{\equal{#4}{\empty}}{}{\AppendKeywordString{#1}{, #4}}%
  \ifthenelse{\equal{#5}{\empty}}{}{\AppendKeywordString{#1}{, #5}}%
  \ifthenelse{\equal{#6}{\empty}}{}{\AppendKeywordString{#1}{, #6}}%
  \ifthenelse{\equal{#7}{\empty}}{}{\AppendKeywordString{#1}{, #7}}%
  \ifthenelse{\equal{#8}{\empty}}{}{\AppendKeywordString{#1}{, #8}}%
  \ifthenelse{\equal{#9}{\empty}}{}{\AppendKeywordString{#1}{, #9}}%
}

% --- 關鍵字 Keyword ---
\def\VarPDFKeywords{\empty} % initialize
\def\GetPDFKeywords{\VarPDFKeywords} % initialize
\def\AppendPDFKeyword#1{\AppendKeywordString{\VarPDFKeywords}{#1}}

\DeclareDocumentCommand{\SetKeywords}{
  m G{\empty} G{\empty}
  G{\empty} G{\empty} G{\empty}
  G{\empty} G{\empty} G{\empty}}
{
  \AppendPDFKeyword{#1}
  \AppendOptionalKeywords{\VarPDFKeywords}{#2}{#3}{#4}{#5}{#6}{#7}{#8}{#9}
} % End of \newcommand{}
